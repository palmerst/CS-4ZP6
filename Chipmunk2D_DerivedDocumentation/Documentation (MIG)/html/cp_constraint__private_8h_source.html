244 static inline cpMat2x2
245 k_tensor(cpBody *a, cpBody *b, cpVect r1, cpVect r2)
246 {
247  cpFloat m_sum = a->CP_PRIVATE(m_inv) + b->CP_PRIVATE(m_inv);
248 
249  // start with Identity*m_sum
250  cpFloat k11 = m_sum, k12 = 0.0f;
251  cpFloat k21 = 0.0f, k22 = m_sum;
252 
253  // add the influence from r1
254  cpFloat a_i_inv = a->CP_PRIVATE(i_inv);
255  cpFloat r1xsq = r1.x * r1.x * a_i_inv;
256  cpFloat r1ysq = r1.y * r1.y * a_i_inv;
257  cpFloat r1nxy = -r1.x * r1.y * a_i_inv;
258  k11 += r1ysq; k12 += r1nxy;
259  k21 += r1nxy; k22 += r1xsq;
260 
261  // add the influnce from r2
262  cpFloat b_i_inv = b->CP_PRIVATE(i_inv);
263  cpFloat r2xsq = r2.x * r2.x * b_i_inv;
264  cpFloat r2ysq = r2.y * r2.y * b_i_inv;
265  cpFloat r2nxy = -r2.x * r2.y * b_i_inv;
266  k11 += r2ysq; k12 += r2nxy;
267  k21 += r2nxy; k22 += r2xsq;
268 
269  // invert
270  cpFloat det = k11*k22 - k12*k21;
271  cpAssertSoft(det != 0.0, "Unsolvable constraint.");
272 
273  cpFloat det_inv = 1.0f / det;
274  return cpMat2x2New(
275  k22*det_inv, -k12*det_inv,
276  -k21*det_inv, k11*det_inv
277  );
278 }
