119 static inline void
120 CircleSegmentQuery(cpShape *shape, cpVect center, cpFloat r1, cpVect a, cpVect b, cpFloat r2, cpSegmentQueryInfo *info)
121 {
122  cpVect da = cpvsub(a, center);
123  cpVect db = cpvsub(b, center);
124  cpFloat rsum = r1 + r2;
125 
126  cpFloat qa = cpvdot(da, da) - 2.0f*cpvdot(da, db) + cpvdot(db, db);
127  cpFloat qb = cpvdot(da, db) - cpvdot(da, da);
128  cpFloat det = qb*qb - qa*(cpvdot(da, da) - rsum*rsum);
129 
130  if (det >= 0.0f){
131  cpFloat t = (-qb - cpfsqrt(det)) / (qa);
132  if (0.0f <= t && t <= 1.0f){
133  cpVect n = cpvnormalize(cpvlerp(da, db, t));
134 
135  info->shape = shape;
136  info->point = cpvsub(cpvlerp(a, b, t), cpvmult(n, r2));
137  info->normal = n;
138  info->alpha = t;
139  }
140  }
141 }
